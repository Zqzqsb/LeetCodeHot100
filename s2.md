class Solution {
public:
    vector<vector<string>> groupAnagrams(vector<string>& strs) {
        // nat tan -> ant
        unordered_map<string , vector<string>> mp;
        
        for(auto& str : strs) {
            string temp = str;
            sort(temp.begin() , temp.end());
            mp[temp].push_back(str);
        }
        
        vector<vector<string>> res;
        for(auto& entry : mp) {
            res.push_back(entry.second);
        }
        return res;
    }
};

class Solution {
public:
    static size_t hashArray(const array<int , 26>&  arr) {
        size_t hashValue = 0;
        std::hash<int> fn;
        
        for(auto i : arr) hashValue = (hashValue << 1) ^ fn(i);
        return hashValue;
    }
    
    vector<vector<string>> groupAnagrams(vector<string>& strs) {
        unordered_map<array<int , 26> , vector<string> , decltype(&Solution::hashArray)> mp(5000 , &Solution::hashArray);
        
        for(auto& str : strs) {
            array<int , 26> arr{}; // 通过字面量手动初始化
            for(auto& c : str) arr[c - 'a'] ++;
            mp[std::move(arr)].push_back(str);
        }
        
        vector<vector<string>> res;
        for(auto& entry : mp) {
           res.emplace_back(std::move(entry.second));
        }
        return res;
    }
};
