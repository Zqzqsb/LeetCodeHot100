class Solution {
public:
    int longestConsecutive(vector<int>& nums) {
        unordered_set<int> zt(nums.begin() , nums.end());

        int res = 0; 
        for(auto& key : zt) {
            if(res > zt.size() / 2) return res;
            if(zt.count(key - 1)) continue;
            int t = key;
            while(zt.count(t + 1)) t ++; // 沿着当前的起点向后查找
            res = max(res , t - key + 1); // 试图更新最大连续序列长度
        }
        return res;
    }
}
